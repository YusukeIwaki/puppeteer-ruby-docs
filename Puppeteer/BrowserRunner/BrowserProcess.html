def initialize(env, executable_path, args)
  stdin, @stdout, @stderr, @thread = Open3.popen3(env, executable_path, *args)
  stdin.close
  @pid = @thread.pid
end

def dispose
  [@stdout, @stderr].each { |io| io.close unless io.closed? }
  @thread.join
end
